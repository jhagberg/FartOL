DNF/MP detection rules (CONTEXT D-12 + RESEARCH §Pattern 5):
- Course has an ordered list of expected control codes ending with a virtual "finish" code (250 by SportIdent convention) or the course's last control.
- A card_read payload has `punches: NdjsonPunch[]`. Punches contain start/check (control_code 1, 2, 3, 4, 5, 240 — depends on dialect) and the regular controls.

Convention LOCKED here for Phase 1:
- The "start" punch control_code is the special value 1 OR 4 (SI convention; some clubs use 5). Filter these out of the matching sequence.
- The "finish" punch control_code is the special value 2 OR 3 (or 240 for newer cards). Filter these out of the matching sequence and treat their presence as "finished".
- The "check" punch (control_code 250 or 0xFE in some encodings) is filtered.
- After filtering, the remaining punches' control_codes are compared in-order against course.controls (also ordered by order_idx).

DNF/MP semantics (CONTEXT D-12 — punch-only):
- If a card_read has no finish punch (after filter): status = 'DNF' (didn't finish).
- If card_read has a finish punch AND all course controls are present in correct order: status = 'OK'.
- Otherwise: status = 'MP', with missing_codes, extra_codes, out_of_order_codes returned for receipt display.
- manual_dnf event sets status='DNF' regardless of punches; payload.reason is shown on receipt.
- un_dnf event reverts: the projection re-runs the dnfMp rules against the latest card_read for that competitor.

Reduce ordering:
- Sort events by (eventTimeMs, localSeq) for determinism. localSeq is the tiebreaker if two events share eventTimeMs.
- A competitor may have multiple card_read events (re-read of the same card during the same competition). Use the latest as the canonical projection; preserve a `card_read_history` array on CompetitorView for the receipt mirror.

Matching (REQ-EVT-CMP-005 D-11):
- On card_read with card_number X: if competitor row in this competition has card_number === X, attach. The competitor row's class_id maps to a course (course.class_id ===); use that course's controls for DNF/MP.
- If no competitor matches AND no card_bound event for this card_number AND this competition: add card_number to pendingUnknownCards. The walk-up modal (plan 14) consumes this list.
- card_bound event creates a NEW competitor row (already done by plan 04 POST /api/competitors), and the reducer removes that card_number from pendingUnknownCards on the next reduce pass.

    ```
    import type { NdjsonPunch } from '@fartol/sportident';

    // Phase 1 punch-filter constants. Documented in this file's header for ADR-style traceability.
    const START_CODES = new Set([1, 4, 5]);       // SI start codes
    const FINISH_CODES = new Set([2, 3, 240]);    // SI finish codes
    const CHECK_CODES = new Set([0, 250, 0xFE]);  // SI check / clear

    export interface StatusResult {
      status: 'OK' | 'MP' | 'DNF';
      missing_codes: number[];
      extra_codes: number[];
      out_of_order_codes: number[];
      elapsed_time_ms: number | null;
      start_time_ms: number | null;
      finish_time_ms: number | null;
    }

    export function detectStatus(punches: readonly NdjsonPunch[], expectedControlCodes: readonly number[]): StatusResult {
      const startPunch = punches.find(p => START_CODES.has(p.control_code));
      const finishPunch = punches.find(p => FINISH_CODES.has(p.control_code));
      const regular = punches.filter(p => !START_CODES.has(p.control_code) &amp;&amp; !FINISH_CODES.has(p.control_code) &amp;&amp; !CHECK_CODES.has(p.control_code));
      const startTimeMs = startPunch?.time_ms ?? null;
      const finishTimeMs = finishPunch?.time_ms ?? null;
      const elapsedTimeMs = (startTimeMs !== null &amp;&amp; finishTimeMs !== null) ? finishTimeMs - startTimeMs : null;
      if (!finishPunch) {
        return { status: 'DNF', missing_codes: [...expectedControlCodes], extra_codes: regular.map(p => p.control_code), out_of_order_codes: [], elapsed_time_ms: null, start_time_ms: startTimeMs, finish_time_ms: null };
      }
      // Order match: expected vs actual, in-order, no skips, no extras.
      const expected = [...expectedControlCodes];
      const actual = regular.map(p => p.control_code);
      const missing: number[] = [];
      const extra: number[] = [];
      const outOfOrder: number[] = [];
      let ei = 0, ai = 0;
      while (ei < expected.length || ai < actual.length) {
        if (ei < expected.length &amp;&amp; ai < actual.length &amp;&amp; expected[ei] === actual[ai]) { ei++; ai++; continue; }
        if (ei < expected.length &amp;&amp; ai < actual.length &amp;&amp; actual.slice(ai).includes(expected[ei]!)) {
          // expected[ei] appears later in actual — current actual[ai] is extra OR out-of-order
          if (expected.includes(actual[ai]!)) outOfOrder.push(actual[ai]!); else extra.push(actual[ai]!);
          ai++;
          continue;
        }
        if (ei < expected.length) { missing.push(expected[ei]!); ei++; continue; }
        if (ai < actual.length) { extra.push(actual[ai]!); ai++; continue; }
      }
      const status = (missing.length === 0 &amp;&amp; extra.length === 0 &amp;&amp; outOfOrder.length === 0) ? 'OK' : 'MP';
      return { status, missing_codes: missing, extra_codes: extra, out_of_order_codes: outOfOrder, elapsed_time_ms: elapsedTimeMs, start_time_ms: startTimeMs, finish_time_ms: finishTimeMs };
    }
    ```

    ```
    import type { Event, Competitor, Course, Class } from '@fartol/shared-types';
    import type { EventPayload } from '../db/schema.ts';
    import type { NdjsonPunch } from '@fartol/sportident';
    import { detectStatus } from './dnfMp.ts';
    import { matchCardToCompetitor } from './matching.ts';
    import type { CompetitionState, CompetitorView, ResultView } from './types.ts';

    export interface ReduceInput {
      competition_id: string;
      events: readonly Event[];
      competitors: readonly Competitor[];
      classes: readonly Class[];
      courses: readonly (Course &amp; { control_codes: readonly number[] })[];   // courses joined with course_controls in order
    }

    export function reduce(input: ReduceInput): CompetitionState {
      // 1. Sort events deterministically.
      const sortedEvents = [...input.events].sort((a, b) => (a.eventTimeMs - b.eventTimeMs) || (a.localSeq - b.localSeq));

      // 2. Build helper maps.
      const courseByClass = new Map<string, Course &amp; { control_codes: readonly number[] }>();
      for (const c of input.courses) if (c.classId) courseByClass.set(c.classId, c);
      const competitorsByCompetition = input.competitors.filter(c => c.competitionId === input.competition_id);

      // Mutable working state for the reduce.
      const competitorViews = new Map<string, CompetitorView>();
      for (const c of competitorsByCompetition) {
        competitorViews.set(c.id, { id: c.id, name: c.name, club: c.club, class_id: c.classId, card_number: c.cardNumber, status: 'PEND',
          card_read_history: [], latest_punches: [], missing_codes: [], extra_codes: [], out_of_order_codes: [], elapsed_time_ms: null, manual_dnf_reason: null });
      }
      const pendingUnknownCards = new Set<number>();
      let lastEventSeq = 0;

      for (const e of sortedEvents) {
        if (e.competitionId !== input.competition_id) continue;  // skip cross-competition noise
        lastEventSeq = Math.max(lastEventSeq, e.localSeq);
        const payload = e.payload as EventPayload;
        switch (payload.event_type) {
          case 'card_read': {
            const competitor = matchCardToCompetitor(payload.card_number, competitorsByCompetition);
            if (!competitor) { pendingUnknownCards.add(payload.card_number); break; }
            const view = competitorViews.get(competitor.id);
            if (!view) break;
            view.card_read_history.push({ event_time_ms: e.eventTimeMs, card_number: payload.card_number, punches: payload.punches });
            view.latest_punches = payload.punches;
            // Look up the course via the competitor's class.
            const course = courseByClass.get(competitor.classId);
            const expected = course ? course.control_codes : [];
            const detected = detectStatus(payload.punches, expected);
            // Only overwrite if not already manually DNF.
            if (view.manual_dnf_reason === null) {
              view.status = detected.status;
              view.missing_codes = detected.missing_codes;
              view.extra_codes = detected.extra_codes;
              view.out_of_order_codes = detected.out_of_order_codes;
              view.elapsed_time_ms = detected.elapsed_time_ms;
            }
            break;
          }
          case 'card_bound': {
            pendingUnknownCards.delete(payload.card_number);
            // The competitor row created by POST /api/competitors is already in `competitors`. Nothing to do here at projection time.
            break;
          }
          case 'manual_dnf': {
            const view = competitorViews.get(payload.competitor_id);
            if (view) { view.status = 'DNF'; view.manual_dnf_reason = payload.reason; }
            break;
          }
          case 'un_dnf': {
            const view = competitorViews.get(payload.competitor_id);
            if (view) {
              view.manual_dnf_reason = null;
              // Re-run dnfMp against latest_punches (if any).
              const competitor = competitorsByCompetition.find(c => c.id === payload.competitor_id);
              const course = competitor ? courseByClass.get(competitor.classId) : undefined;
              const expected = course ? course.control_codes : [];
              if (view.latest_punches.length > 0) {
                const detected = detectStatus(view.latest_punches, expected);
                view.status = detected.status;
                view.missing_codes = detected.missing_codes; view.extra_codes = detected.extra_codes; view.out_of_order_codes = detected.out_of_order_codes; view.elapsed_time_ms = detected.elapsed_time_ms;
              } else { view.status = 'PEND'; view.missing_codes = []; view.extra_codes = []; view.out_of_order_codes = []; view.elapsed_time_ms = null; }
            }
            break;
          }
          // card_inserted, card_removed, frame_error, connection_changed do not change projection state.
        }
      }

      // 3. Build results_by_class.
      const resultsByClass = new Map<string, ResultView[]>();
      for (const cls of input.classes) {
        const inClass: CompetitorView[] = [];
        for (const v of competitorViews.values()) if (v.class_id === cls.id) inClass.push(v);
        // Sort: OK first by elapsed_time_ms ASC; MP next; DNF next; PEND last. Stable sort.
        const order = { OK: 0, MP: 1, DNF: 2, PEND: 3 };
        inClass.sort((a, b) => order[a.status] - order[b.status] || ((a.elapsed_time_ms ?? Number.MAX_SAFE_INTEGER) - (b.elapsed_time_ms ?? Number.MAX_SAFE_INTEGER)) || a.name.localeCompare(b.name));
        // Assign place + behind_leader_ms within OK group.
        let place = 0;
        let leaderTime: number | null = null;
        const rows: ResultView[] = inClass.map(v => {
          let p: number | null = null;
          let behind: number | null = null;
          if (v.status === 'OK' &amp;&amp; v.elapsed_time_ms !== null) {
            place++;
            p = place;
            if (leaderTime === null) leaderTime = v.elapsed_time_ms;
            behind = v.elapsed_time_ms - leaderTime;
          }
          return { competitor_id: v.id, name: v.name, club: v.club, status: v.status, elapsed_time_ms: v.elapsed_time_ms, place: p, behind_leader_ms: behind };
        });
        resultsByClass.set(cls.id, rows);
      }

      return {
        competition_id: input.competition_id,
        competitors: competitorViews,
        results_by_class: resultsByClass,
        pending_unknown_cards: [...pendingUnknownCards].sort((a, b) => a - b),
        last_event_seq: lastEventSeq,
      };
    }
    ```
