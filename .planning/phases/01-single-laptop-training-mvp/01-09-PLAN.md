Auto-bind contract:
- After EntryList import succeeds (plan 05), the new competitors carry cardNumber values that may already have been read while the bridge was idle.
- autoBindNewCompetitors walks events.competitionId = competitionId, eventType = 'card_read', payload.card_number IN (set of new competitors.cardNumber values), and emits a synthetic card_bound event for each MATCH that doesn't already have one.
- Synthetic card_bound payload: { event_type: 'card_bound', competitor_id, card_number, walkup: false, consent_at_ms: <competitor.consent_at_ms> } — walkup flag distinguishes from POST /api/competitors path.
- markDirty after autoBindNewCompetitors → projection runs → pending_unknown_cards now excludes those cards.

Race semantics:

- card_read (cardNumber=X) at t=0, no competitor → pending_unknown_cards has X.
- EntryList import at t=5s creates competitor with cardNumber=X → autoBindNewCompetitors emits card_bound(t=5s) → projection re-runs → X removed from pending.
- Crucially: the original card_read at t=0 is now retroactively attached to the competitor because matchCardToCompetitor finds them.

    export type CardIndex = ReadonlyMap<number, Competitor>;
    export function buildCardIndex(competitors: readonly Competitor[]): CardIndex {
      const map = new Map<number, Competitor>();
      for (const c of competitors) if (c.cardNumber !== null) map.set(c.cardNumber, c);

      return null;
    }

    // Authored for fartola. Not ported from upstream.
    // Retroactive card-to-competitor binding. After EntryList import (plan 05), prior card_read
    // events that match the new competitors.cardNumber values get synthetic card_bound events
    // appended to the log. Plan 09 closes the race between bridge readout and entry import.
    import { eq, and, sql, inArray } from 'drizzle-orm';
    import { events, competitors as competitorsTable } from '../db/schema.ts';
    import { insertEvent } from '../si/eventInserter.ts';
    import type { DbHandle } from '../db/index.ts';

    export interface AutoBindResult { bound: Array<{ competitor_id: string; card_number: number }>; }

    export function autoBindNewCompetitors(handle: DbHandle, competitionId: string, nodeId: string): AutoBindResult {
      // 1. Find competitors in this competition with non-null cardNumber and no existing card_bound event.
      const allComp = handle.db.select().from(competitorsTable).where(eq(competitorsTable.competitionId, competitionId)).all();
      const withCard = allComp.filter(c => c.cardNumber !== null);
      // 2. For each, check whether a card_bound event already exists.
      const result: AutoBindResult = { bound: [] };
      for (const c of withCard) {
        const existing = handle.db.select({ id: events.localSeq })
          .from(events)
          .where(and(eq(events.competitionId, competitionId), eq(events.eventType, 'card_bound'), sql`json_extract(${events.payload}, '$.competitor_id') = ${c.id}`))
          .get();
        if (existing) continue;
        // Only emit if there's at least one card_read for this card_number — otherwise no race to resolve.
        const seenRead = handle.db.select({ id: events.localSeq })
          .from(events)
          .where(and(eq(events.competitionId, competitionId), eq(events.eventType, 'card_read'), sql`json_extract(${events.payload}, '$.card_number') = ${c.cardNumber}`))
          .get();
        if (!seenRead) continue;
        // Emit synthetic card_bound. event_time_ms = consent_at_ms (when competitor was created), which is > earlier card_read time.
        insertEvent(handle, nodeId, 'card_bound', c.consentAtMs, {
          event_type: 'card_bound',
          competitor_id: c.id,
          card_number: c.cardNumber!,
          walkup: false,
          consent_at_ms: c.consentAtMs,
        }, competitionId);
        result.bound.push({ competitor_id: c.id, card_number: c.cardNumber! });
      }
      return result;
    }
    ```

    import { eq, and, desc } from 'drizzle-orm';
    import { events, competitors as competitorsTable } from '../db/schema.ts';

    export default async function readoutRoutes(app: FastifyInstance) {
      app.get('/api/competitions/:id/readout', async (req, reply) => {
        const competitionId = (req.params as { id: string }).id;
        // Pull last 12 card_read events.
        const cardReadRows = app.fartolaDb.db.select()
          .from(events)
          .where(and(eq(events.competitionId, competitionId), eq(events.eventType, 'card_read')))
          .orderBy(desc(events.eventTimeMs), desc(events.localSeq))
          .limit(12)
          .all();

        // Build cardIndex once for matching.
        const compRows = app.fartolaDb.db.select().from(competitorsTable).where(eq(competitorsTable.competitionId, competitionId)).all();
        const byCard = new Map<number, typeof compRows[number]>();
        for (const c of compRows) if (c.cardNumber !== null) byCard.set(c.cardNumber, c);

        // For history, derive status per-event by running the projection (or fall back to per-event detectStatus + the competitor's class).
        // Cheaper approach: hit projection store and read pre-computed competitor views.
        let projection = app.projectionStore.get(competitionId);
        if (!projection) projection = app.projectionStore.recomputeNow(competitionId);

        const history = cardReadRows.map(e => {
          const payload = e.payload as { event_type: 'card_read'; card_number: number; card_type: string; punches: unknown[] };
          const competitor = byCard.get(payload.card_number);
          const view = competitor ? projection?.competitors.get(competitor.id) : undefined;
          return {
            event_time_ms: e.eventTimeMs,
            local_seq: e.localSeq,
            card_number: payload.card_number,
            card_type: payload.card_type,
            competitor_id: competitor?.id ?? null,
            competitor_name: competitor?.name ?? null,
            status: view?.status ?? 'PEND',
            unmatched: !competitor,
          };
        });

        const currentRead = history[0] ?? null;
        const pendingUnknownCards = projection?.pending_unknown_cards ?? [];
        return reply.send({
          competition_id: competitionId,
          active: app.activeCompetitionId === competitionId,
          current_read: currentRead,
          history,
          pending_unknown_cards: pendingUnknownCards,
        });
      });
    }
    ```
