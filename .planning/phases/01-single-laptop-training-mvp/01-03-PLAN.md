**WS protocol contract (LOCKED for the entire phase; per-channel-kind branch per C-M1):**

- Client connects to ws://localhost:3000/ws.
- Client sends `{ type: 'hello', channels: [...], last_seen_seq: N }` immediately.
- Server iterates the requested channels and dispatches per channel-kind:
  - For `readout:<id>` channels: server replays missed events via `{ type: 'replay', channel: 'readout:<id>', payload: EventDTO, seq: M }` envelopes for each event with local_seq > N. RAW EVENT REPLAY IS EXCLUSIVE TO `readout:` CHANNELS.
  - For `results:<id>` channels: server emits NO raw `replay` envelopes. The `results_full` envelope is the sole reply (plan 08 wires the actual emission; this plan locks the branch and asserts via test that `results:` hello produces zero `replay` envelopes).
- Server then sends live envelopes as events arrive: `{ type: 'card_read', channel: 'readout:<id>', payload: ..., seq: M }` on readout channels; `{ type: 'results_update' | 'results_full', channel: 'results:<id>', payload: ..., seq: M }` on results channels.
- Client can send `{ type: 'subscribe', channel: 'results:<id>' }` at any time.

Plan 08 fills in the actual `results_full` emission body. Plan 03 ships the dispatch skeleton with a stub: when a `results:` hello arrives, the handler currently logs a notice ("plan-08 results_full not yet wired") and emits NOTHING (no `replay` envelopes, no fabricated `results_full`). The contract — never emit a `replay` envelope on a results: channel — is what this plan locks. Plan 08 lifts the stub to a real `results_full` payload.

    apps/edge/src/ws/replay.ts — SQLite-backed missed-event replay (T-EVENT-REPLAY mitigation). C-M1: callers must check channelKind === 'readout' BEFORE invoking replayChannel; the function itself stays kind-agnostic at the data layer but the WS handler branches above it:
    ```
    import { events } from '../db/schema.ts';
    import { gt, eq, and, sql } from 'drizzle-orm';
    import type { DbHandle } from '../db/index.ts';
    import type { ChannelName } from './channels.ts';

    export function parseChannel(channel: ChannelName): { kind: 'readout' | 'results'; competitionId: string } {
      const idx = channel.indexOf(':');
      return { kind: channel.slice(0, idx) as 'readout' | 'results', competitionId: channel.slice(idx + 1) };
    }

    export function maxLocalSeq(handle: DbHandle, nodeId: string): number {
      const row = handle.db.select({ max: sql<number>`coalesce(max(${events.localSeq}), 0)` })
        .from(events).where(eq(events.nodeId, nodeId)).get();
      return row?.max ?? 0;
    }

    export function replayChannel(handle: DbHandle, channel: ChannelName, lastSeenSeq: number, nodeId: string): Array<{ seq: number; payload: unknown; event_type: string }> {
      // Validate seq sanity (T-EVENT-REPLAY): negative OR > max+1 -> empty result.
      const max = maxLocalSeq(handle, nodeId);
      if (lastSeenSeq < 0 || lastSeenSeq > max + 1) return [];
      const { competitionId } = parseChannel(channel);
      // Pull events for this competition with local_seq > lastSeenSeq, ordered.
      const rows = handle.db.select().from(events)
        .where(and(eq(events.competitionId, competitionId), eq(events.nodeId, nodeId), gt(events.localSeq, lastSeenSeq)))
        .orderBy(events.localSeq).all();
      return rows.map(r => ({ seq: r.localSeq, payload: r.payload, event_type: r.eventType }));
    }
    ```

    ```
    import type { ChannelName, WsEnvelope, WsHelloMessage, WsSubscribeMessage } from '@fartola/shared-types';
    export class WsClient {
      private ws: WebSocket | null = null;
      private channels = new Set<ChannelName>();
      private lastSeenSeq = 0;
      private backoff = [1000, 2000, 4000, 8000, 16000, 30000];
      private attempt = 0;
      private closed = false;
      constructor(private url: string, private onMessage: (env: WsEnvelope) => void) {}
      connect(): void {
        if (this.closed) return;
        this.ws = new WebSocket(this.url);
        this.ws.onopen = () => {
          this.attempt = 0;
          const hello: WsHelloMessage = { type: 'hello', channels: [...this.channels], last_seen_seq: this.lastSeenSeq };
          this.ws!.send(JSON.stringify(hello));
        };
        this.ws.onmessage = (ev) => {
          try {
            const env = JSON.parse(ev.data) as WsEnvelope &amp; { seq?: number };
            if (typeof env.seq === 'number' &amp;&amp; env.seq > this.lastSeenSeq) this.lastSeenSeq = env.seq;
            this.onMessage(env);
          } catch { /* malformed envelope - skip */ }
        };
        this.ws.onclose = () => {
          if (this.closed) return;
          const delay = this.backoff[Math.min(this.attempt, this.backoff.length - 1)]!;
          this.attempt++;
          setTimeout(() => this.connect(), delay);
        };
      }
      subscribe(channel: ChannelName): void {
        this.channels.add(channel);
        if (this.ws?.readyState === 1) {
          const msg: WsSubscribeMessage = { type: 'subscribe', channel };
          this.ws.send(JSON.stringify(msg));
        }
      }
      close(): void { this.closed = true; this.ws?.close(); }
    }
    ```
