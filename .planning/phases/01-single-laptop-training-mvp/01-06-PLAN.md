    ```
    import { events } from '../db/schema.ts';
    import { nextLocalSeq } from '../db/seq.ts';
    import type { DbHandle } from '../db/index.ts';
    import type { EventPayload } from '../db/schema.ts';

    export interface InsertResult { node_id: string; local_seq: number; event_time_ms: number; recorded_at_ms: number; }

    export function insertEvent(
      handle: DbHandle,
      nodeId: string,
      eventType: EventPayload['event_type'],
      eventTimeMs: number,
      payload: EventPayload,
      competitionId: string | null,
    ): InsertResult {
      const recordedAtMs = Date.now();
      let localSeq = 0;
      handle.sqlite.transaction(() => {
        localSeq = nextLocalSeq(handle, nodeId);
        handle.db.insert(events).values({ nodeId, localSeq, competitionId, eventType, eventTimeMs, recordedAtMs, payload }).run();
      })();
      return { node_id: nodeId, local_seq: localSeq, event_time_ms: eventTimeMs, recorded_at_ms: recordedAtMs };
    }
    ```

    apps/edge/src/si/bridge.ts — the adapter. The payload shape MUST mirror NdjsonEmitter (packages/sportident/src/output/ndjson.ts line 253) so the events.payload column shape is identical across the SI bridge and the NDJSON CLI:

    ```
    // Authored for fartol. Not ported from upstream.
    // Wires Phase 0's SiMainStation events into the Phase 1 SQLite event log + WS broadcast.
    // Per .planning/phases/01-single-laptop-training-mvp/01-RESEARCH.md §"Code Examples — Wire SI events"
    // and .planning/phases/01-single-laptop-training-mvp/01-PATTERNS.md §"apps/edge/src/si/bridge.ts".
    // Punches surface frozen by packages/sportident/src/output/ndjson.ts line 253:
    //   const punches = (raceResult.punches ?? []).map(...);
    // The bridge mirrors that path EXACTLY. cardHolder is competitor metadata, not punches.
    import type { SiMainStation, BaseSiCard, FrameError, ConnectionState, NdjsonPunch } from '@fartol/sportident';
    import { cardTypeFromNumber } from './cardType.ts';
    import { insertEvent } from './eventInserter.ts';
    import { readoutChannel } from '@fartol/shared-types';
    import type { DbHandle } from '../db/index.ts';
    import type { ChannelName } from '../ws/channels.ts';

    export interface BridgeOpts {
      handle: DbHandle;
      nodeId: string;
      getActiveCompetitionId: () => string | null;
      broadcast: (channel: ChannelName, envelope: { type: string; payload: unknown; seq?: number }) => void;
    }

    export interface AttachedBridge { detach: () => void; }

    export function attachBridge(station: SiMainStation, opts: BridgeOpts): AttachedBridge {
      // Helper: broadcast only when an active competition is set. With no active
      // competition, the event still persists (with competition_id=null) but NO
      // WS broadcast fires — there is no 'readout:__idle__' channel; nothing
      // subscribes to such a string. See must_haves.truths.
      function maybeBroadcast(type: string, payload: unknown, seq: number) {
        const activeId = opts.getActiveCompetitionId();
        if (activeId === null) return;
        opts.broadcast(readoutChannel(activeId), { type, payload, seq });
      }

      const handlers = {
        cardInserted: (card: BaseSiCard) => {
          const activeId = opts.getActiveCompetitionId();
          const payload = { event_type: 'card_inserted' as const, card_number: card.cardNumber, card_type: cardTypeFromNumber(card.cardNumber) };
          const r = insertEvent(opts.handle, opts.nodeId, 'card_inserted', Date.now(), payload, activeId);
          maybeBroadcast('card_inserted', payload, r.local_seq);
        },
        cardRead: (card: BaseSiCard) => {
          // LOCKED punches path — sources from card.raceResult.punches per NdjsonEmitter line 253.
          // Reading card.cardHolder?.punches is WRONG (B-1 from plan-check revision): cardHolder
          // carries first_name/last_name/club/email/zip metadata, NOT punches. With the wrong path,
          // every real card_read inserts empty punches, the plan-07 reducer marks every OK card
          // as DNF, and SC#3 fails at the StorTuna bench.
          const punches = (card.raceResult as { punches?: NdjsonPunch[] } | undefined)?.punches ?? [];
          const activeId = opts.getActiveCompetitionId();
          const payload = { event_type: 'card_read' as const, card_number: card.cardNumber, card_type: cardTypeFromNumber(card.cardNumber), punches };
          const r = insertEvent(opts.handle, opts.nodeId, 'card_read', Date.now(), payload, activeId);
          maybeBroadcast('card_read', payload, r.local_seq);
        },
        cardRemoved: (cardNumber: number) => {
          const activeId = opts.getActiveCompetitionId();
          const payload = { event_type: 'card_removed' as const, card_number: cardNumber };
          const r = insertEvent(opts.handle, opts.nodeId, 'card_removed', Date.now(), payload, activeId);
          maybeBroadcast('card_removed', payload, r.local_seq);
        },
        frameError: (err: FrameError) => {
          const activeId = opts.getActiveCompetitionId();
          const payload = { event_type: 'frame_error' as const, reason: err.code, raw: err.frame_hex ?? '' };
          const r = insertEvent(opts.handle, opts.nodeId, 'frame_error', Date.now(), payload, activeId);
          maybeBroadcast('frame_error', payload, r.local_seq);
        },
        connectionChanged: (state: ConnectionState, error?: Error) => {
          const activeId = opts.getActiveCompetitionId();
          const payload = { event_type: 'connection_changed' as const, state, ...(error ? { error: error.message } : {}) };
          const r = insertEvent(opts.handle, opts.nodeId, 'connection_changed', Date.now(), payload, activeId);
          maybeBroadcast('connection_changed', payload, r.local_seq);
        },
      };
      station.on('cardInserted', handlers.cardInserted);
      station.on('cardRead', handlers.cardRead);
      station.on('cardRemoved', handlers.cardRemoved);
      station.on('frameError', handlers.frameError);
      station.on('connectionChanged', handlers.connectionChanged);
      return {
        detach() {
          station.off('cardInserted', handlers.cardInserted);
          station.off('cardRead', handlers.cardRead);
          station.off('cardRemoved', handlers.cardRemoved);
          station.off('frameError', handlers.frameError);
          station.off('connectionChanged', handlers.connectionChanged);
        },
      };
    }
    ```
