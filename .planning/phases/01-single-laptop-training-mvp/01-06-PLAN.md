    import { events } from '../db/schema.ts';
    import { nextLocalSeq } from '../db/seq.ts';
    import type { DbHandle } from '../db/index.ts';
    import type { EventPayload } from '../db/schema.ts';

    export interface InsertResult { node_id: string; local_seq: number; event_time_ms: number; recorded_at_ms: number; }

    export function insertEvent(
      handle: DbHandle,
      nodeId: string,
      eventType: EventPayload['event_type'],
      eventTimeMs: number,
      payload: EventPayload,
      competitionId: string | null,
    ): InsertResult {
      const recordedAtMs = Date.now();
      let localSeq = 0;
      handle.sqlite.transaction(() => {
        localSeq = nextLocalSeq(handle, nodeId);
        handle.db.insert(events).values({ nodeId, localSeq, competitionId, eventType, eventTimeMs, recordedAtMs, payload }).run();
      })();
      return { node_id: nodeId, local_seq: localSeq, event_time_ms: eventTimeMs, recorded_at_ms: recordedAtMs };
    }
    ```

    // Authored for fartola. Not ported from upstream.
    // Builds the events.payload column body for `card_read` events. Mirrors
    // NdjsonEmitter.card_read EXACTLY (packages/sportident/src/output/ndjson.ts
    // lines 241-296). The function's return type is Phase 0's `CardReadEvent`
    // import; any future shape drift in Phase 0 surfaces as a TS compile error
    // here (T-PAYLOAD-DRIFT mitigation, codex C-H2 finding).
    //
    // Per codex review C-H2: the prior revision dropped start/finish/check/clear
    // and card_holder. Plan 07 reducer then guessed at start/finish by punch code,
    // which is wrong: those special codes are control-station punches in Phase 0's
    // surface, not start/finish events. Locking the full shape here resolves the
    // chain (plan 07 reducer reads payload.start + payload.finish directly).
    import type { BaseSiCard, CardReadEvent, NdjsonPunch, HalfDayClock } from '@fartola/sportident';
    import { toHalfDayClock } from '@fartola/sportident';   // verified exported in Task 1 setup
    import { cardTypeFromNumber } from './cardType.ts';

    // snakeCaseKeys: a one-level snake_case conversion of an arbitrary record.
    // Mirrors NdjsonEmitter's helper (packages/sportident/src/output/ndjson.ts).
    function snakeCaseKeys(o: Record<string, unknown>): Record<string, unknown> {
      const out: Record<string, unknown> = {};
      for (const [k, v] of Object.entries(o)) {
        out[k.replace(/[A-Z]/g, (m) => '_' + m.toLowerCase())] = v;
      }
      return out;
    }

    /**
     * Construct a `card_read` events.payload body from a Phase 0 BaseSiCard
     * instance. Return type is Phase 0's `CardReadEvent` so TS surfaces any
     * drift between Phase 0 and Phase 1 at compile time.
     *
     * Note: the returned shape omits the NdjsonEmitter `_base()` fields
     * (schema_version, ts, device_path, device_serial) — those are NDJSON-
     * stream-specific. The events row already carries equivalent metadata
     * (recorded_at_ms, node_id). The discriminant `event_type: 'card_read'`
     * is included so the EventPayload union narrows correctly.
     */
    export function buildCardReadPayload(card: BaseSiCard): Omit<CardReadEvent, 'schema_version' | 'ts' | 'device_path' | 'device_serial'> {
      const raceResult = card.raceResult;
      const c = card as BaseSiCard &amp; { punchCount?: number; uid?: number };
      const punches: NdjsonPunch[] = (raceResult.punches ?? []).map((p) => {
        const clock = toHalfDayClock(p.time);
        return {
          code: p.code,
          seconds_in_half_day: clock?.seconds_in_half_day ?? 0,
          half_day: clock?.half_day ?? 0,
          weekday: clock?.weekday ?? null,
        };
      });
      const cardNumber = raceResult.cardNumber ?? card.cardNumber;
      const out: Omit<CardReadEvent, 'schema_version' | 'ts' | 'device_path' | 'device_serial'> = {
        event_type: 'card_read',
        card_type: cardTypeFromNumber(cardNumber),
        card_number: cardNumber,
        start: toHalfDayClock(raceResult.startTime),
        finish: toHalfDayClock(raceResult.finishTime),
        check: toHalfDayClock(raceResult.checkTime),
        clear: toHalfDayClock(raceResult.clearTime),
        punch_count: c.punchCount ?? punches.length,
        punches,
        card_holder: raceResult.cardHolder === undefined
          ? null
          : snakeCaseKeys(raceResult.cardHolder as Record<string, unknown>),
      };
      if (card.cardSeriesByte !== undefined) (out as CardReadEvent).card_series_byte = card.cardSeriesByte;
      if (c.uid !== undefined) (out as CardReadEvent).uid = c.uid;
      return out;
    }
    ```

    import type { SiMainStation, BaseSiCard, FrameError, ConnectionState } from '@fartola/sportident';
    import { cardTypeFromNumber } from './cardType.ts';
    import { buildCardReadPayload } from './cardReadPayload.ts';
    import { insertEvent } from './eventInserter.ts';
    import { readoutChannel } from '@fartola/shared-types';
    import type { DbHandle } from '../db/index.ts';
    import type { ChannelName } from '../ws/channels.ts';

    export interface BridgeOpts {
      handle: DbHandle;
      nodeId: string;
      getActiveCompetitionId: () => string | null;
      broadcast: (channel: ChannelName, envelope: { type: string; payload: unknown; seq?: number }) => void;
    }

    export interface AttachedBridge { detach: () => void; }

    export function attachBridge(station: SiMainStation, opts: BridgeOpts): AttachedBridge {
      function maybeBroadcast(type: string, payload: unknown, seq: number) {
        const activeId = opts.getActiveCompetitionId();
        if (activeId === null) return;
        opts.broadcast(readoutChannel(activeId), { type, payload, seq });
      }

      const handlers = {
        cardInserted: (card: BaseSiCard) => {
          const activeId = opts.getActiveCompetitionId();
          const payload = { event_type: 'card_inserted' as const, card_number: card.cardNumber, card_type: cardTypeFromNumber(card.cardNumber) };
          const r = insertEvent(opts.handle, opts.nodeId, 'card_inserted', Date.now(), payload, activeId);
          maybeBroadcast('card_inserted', payload, r.local_seq);
