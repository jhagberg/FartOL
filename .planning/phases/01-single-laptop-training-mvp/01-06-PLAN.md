    apps/edge/src/si/eventInserter.ts — single insertion path used by both the SI bridge (this plan) AND the simulate-read dev endpoint (plan 03 — Task 3 below refactors dev.ts to call insertEvent):
    ```
    import { events } from '../db/schema.ts';
    import { nextLocalSeq } from '../db/seq.ts';
    import type { DbHandle } from '../db/index.ts';
    import type { EventPayload } from '../db/schema.ts';

    export interface InsertResult { node_id: string; local_seq: number; event_time_ms: number; recorded_at_ms: number; }

    export function insertEvent(
      handle: DbHandle,
      nodeId: string,
      eventType: EventPayload['event_type'],
      eventTimeMs: number,
      payload: EventPayload,
      competitionId: string | null,
    ): InsertResult {
      const recordedAtMs = Date.now();
      let localSeq = 0;
      handle.sqlite.transaction(() => {
        localSeq = nextLocalSeq(handle, nodeId);
        handle.db.insert(events).values({ nodeId, localSeq, competitionId, eventType, eventTimeMs, recordedAtMs, payload }).run();
      })();
      return { node_id: nodeId, local_seq: localSeq, event_time_ms: eventTimeMs, recorded_at_ms: recordedAtMs };
    }
    ```

    apps/edge/src/si/bridge.ts — the adapter. RESEARCH §"Code Examples" + PATTERNS analog:
    ```
    // Authored for fartol. Not ported from upstream.
    // Wires Phase 0's SiMainStation events into the Phase 1 SQLite event log + WS broadcast.
    // Per .planning/phases/01-single-laptop-training-mvp/01-RESEARCH.md §"Code Examples — Wire SI events"
    // and .planning/phases/01-single-laptop-training-mvp/01-PATTERNS.md §"apps/edge/src/si/bridge.ts".
    import type { SiMainStation, BaseSiCard, FrameError, ConnectionState } from '@fartol/sportident';
    import { cardTypeFromNumber } from './cardType.ts';
    import { insertEvent } from './eventInserter.ts';
    import { readoutChannel } from '@fartol/shared-types';
    import type { DbHandle } from '../db/index.ts';
    import type { ChannelName } from '../ws/channels.ts';

    export interface BridgeOpts {
      handle: DbHandle;
      nodeId: string;
      getActiveCompetitionId: () => string | null;
      broadcast: (channel: ChannelName, envelope: { type: string; payload: unknown; seq?: number }) => void;
    }

    export interface AttachedBridge { detach: () => void; }

    export function attachBridge(station: SiMainStation, opts: BridgeOpts): AttachedBridge {
      const handlers = {
        cardInserted: (card: BaseSiCard) => {
          const payload = { event_type: 'card_inserted' as const, card_number: card.cardNumber, card_type: cardTypeFromNumber(card.cardNumber) };
          const r = insertEvent(opts.handle, opts.nodeId, 'card_inserted', Date.now(), payload, opts.getActiveCompetitionId());
          opts.broadcast(readoutChannel(opts.getActiveCompetitionId() ?? '__idle__'), { type: 'card_inserted', payload, seq: r.local_seq });
        },
        cardRead: (card: BaseSiCard) => {
          const punches = card.cardHolder?.punches ?? [];
          const payload = { event_type: 'card_read' as const, card_number: card.cardNumber, card_type: cardTypeFromNumber(card.cardNumber), punches };
          const r = insertEvent(opts.handle, opts.nodeId, 'card_read', Date.now(), payload, opts.getActiveCompetitionId());
          opts.broadcast(readoutChannel(opts.getActiveCompetitionId() ?? '__idle__'), { type: 'card_read', payload, seq: r.local_seq });
        },
        cardRemoved: (cardNumber: number) => {
          const payload = { event_type: 'card_removed' as const, card_number: cardNumber };
          const r = insertEvent(opts.handle, opts.nodeId, 'card_removed', Date.now(), payload, opts.getActiveCompetitionId());
          opts.broadcast(readoutChannel(opts.getActiveCompetitionId() ?? '__idle__'), { type: 'card_removed', payload, seq: r.local_seq });
        },
        frameError: (err: FrameError) => {
          const payload = { event_type: 'frame_error' as const, reason: err.code, raw: err.frame_hex ?? '' };
          const r = insertEvent(opts.handle, opts.nodeId, 'frame_error', Date.now(), payload, opts.getActiveCompetitionId());
          opts.broadcast(readoutChannel(opts.getActiveCompetitionId() ?? '__idle__'), { type: 'frame_error', payload, seq: r.local_seq });
        },
        connectionChanged: (state: ConnectionState, error?: Error) => {
          const payload = { event_type: 'connection_changed' as const, state, ...(error ? { error: error.message } : {}) };
          const r = insertEvent(opts.handle, opts.nodeId, 'connection_changed', Date.now(), payload, opts.getActiveCompetitionId());
          opts.broadcast(readoutChannel(opts.getActiveCompetitionId() ?? '__idle__'), { type: 'connection_changed', payload, seq: r.local_seq });
        },
      };
      station.on('cardInserted', handlers.cardInserted);
      station.on('cardRead', handlers.cardRead);
      station.on('cardRemoved', handlers.cardRemoved);
      station.on('frameError', handlers.frameError);
      station.on('connectionChanged', handlers.connectionChanged);
      return {
        detach() {
          station.off('cardInserted', handlers.cardInserted);
          station.off('cardRead', handlers.cardRead);
          station.off('cardRemoved', handlers.cardRemoved);
          station.off('frameError', handlers.frameError);
          station.off('connectionChanged', handlers.connectionChanged);
        },
      };
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| /dev/ttyUSB0 → SerialTransport | Trust kernel; serialport already validated in Phase 0 |
| station events → events table | Single insertEvent path with sqlite.transaction; sequence integrity guarded by composite PK |
| POST /api/sessions/active-competition → bridge routing | Operator action; routes future events to chosen competition. Authenticated by localhost-only binding (plan 01 T-WS-FAN-OUT mitigation). |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-SEQ-COLLISION | Tampering | events PK (node_id, local_seq) | mitigate | nextLocalSeq + insert inside sqlite.transaction; SQLite serializes transactions in WAL mode. Test 2 in eventInserter.test.ts. |
| T-EVENT-REPLAY-DUPLICATE | Tampering | PlaybackTransport (dev tool) | accept | --bench-replay flag is operator-controlled; events table dedupes by composite PK. Replaying the same fixture twice would attempt to insert duplicate (node_id, local_seq) only if seq counter was reset; current design uses live counter so duplicates are impossible. |
| T-USB-EBUSY | DoS | transport.open() | mitigate | Reconnect chain 250ms/500ms/1s/2s/5s (RESEARCH Pitfall 4); operator can POST /api/sessions/reconnect-bridge to retry. |
| T-EVENT-DROP-ON-CRASH | Reliability | Bridge process exit | mitigate | sqlite WAL guarantees committed events survive. The transaction in insertEvent is synchronous + atomic. Process restart resumes from last event in log per REQ-OPS-002. |
| T-COMP-LEAK | Privacy | Bridge routes events to active competition | accept | Operator-controlled toggle; mistaking competition assignment leaks reads across competitions. Mitigation deferred to Phase 2 (audit log + ops dashboard REQ-OPS-004). |
</threat_model>
