    ```
    export async function confirmConsent(competitorId: string): Promise<{ ok: true } | { ok: false; status: number; data: { error: string; message?: string } }> {
      const res = await fetch(\`/api/competitors/\${encodeURIComponent(competitorId)}\`, {
        method: 'PATCH',
        headers: { 'content-type': 'application/json' },
        body: JSON.stringify({ consent_status: 'confirmed_on_read', consent_at_ms: Date.now() }),
      });
      if (res.status === 200) return { ok: true };
      const data = await res.json().catch(() => ({ error: 'unknown' }));
      return { ok: false, status: res.status, data };
    }
    ```

    const PatchConsentSchema = z.object({
      consent_status: z.literal('confirmed_on_read'),
      consent_at_ms: z.number().int().positive(),
    });
    app.patch('/api/competitors/:id', async (req, reply) => {
      const id = (req.params as { id: string }).id;
      const parsed = PatchConsentSchema.safeParse(req.body);
      if (!parsed.success) return reply.code(400).send({ error: 'bad_request', issues: parsed.error.issues });
      const row = app.fartolDb.db.select().from(competitors).where(eq(competitors.id, id)).get();
      if (!row) return reply.code(404).send({ error: 'competitor_not_found' });
      if (row.consentStatus !== 'pending_first_read') return reply.code(422).send({ error: 'consent_not_pending', current: row.consentStatus });
      app.fartolDb.sqlite.transaction(() => {
        app.fartolDb.db.update(competitors)
          .set({ consentStatus: 'confirmed_on_read', consentAtMs: parsed.data.consent_at_ms })
          .where(eq(competitors.id, id)).run();
        const seq = nextLocalSeq(app.fartolDb, app.fartolNodeId);
        app.fartolDb.db.insert(events).values({
          nodeId: app.fartolNodeId, localSeq: seq,
          competitionId: row.competitionId,
          eventType: 'consent_confirmed',
          eventTimeMs: Date.now(), recordedAtMs: Date.now(),
          payload: { event_type: 'consent_confirmed', competitor_id: id, prior_consent_status: 'pending_first_read' },
        }).run();
      })();
      app.projectionStore.markDirty(row.competitionId);
      return reply.code(200).send({ ok: true, competitor_id: id });
    });
