    import { XMLParser } from 'fast-xml-parser';

    export type ParsedXml =
      | { kind: 'CourseData'; data: ParsedCourseData }
      | { kind: 'EntryList'; data: ParsedEntryList };

    const parser = new XMLParser({
      ignoreAttributes: false,
      attributeNamePrefix: '@',
      processEntities: false,         // T-FILE-IMPORT: disable entity expansion (billion-laughs)
      allowBooleanAttributes: true,
      // fast-xml-parser does NOT support DTD; explicitly reject any input containing <!DOCTYPE
    });

    export function parseIofXml(xmlSource: string): ParsedXml {
      // Pre-flight: reject DOCTYPE and external entity declarations.
      if (/<!DOCTYPE/i.test(xmlSource)) throw new Error('DOCTYPE not allowed');
      if (/<!ENTITY/i.test(xmlSource)) throw new Error('ENTITY declarations not allowed');
      const parsed = parser.parse(xmlSource);
      const rootKey = Object.keys(parsed).find(k => k !== '?xml' &amp;&amp; k !== '@_xmlns');
      if (!rootKey) throw new Error('No XML root element');
      if (rootKey === 'CourseData') return { kind: 'CourseData', data: normalizeCourseData(parsed.CourseData) };
      if (rootKey === 'EntryList') return { kind: 'EntryList', data: normalizeEntryList(parsed.EntryList) };
      throw new Error(`Unsupported XML root element: ${rootKey}. Expected CourseData or EntryList.`);
    }
    ```

    // Authored for fartol. Not ported from upstream.
    // XSD validator wrapping libxmljs2-xsd (or xmllint-wasm fallback). Parses the
    // bundled IOF.xsd once at module load; reuse on every call. RESEARCH §Pattern 7.
    import path from 'node:path';
    import { fileURLToPath } from 'node:url';
    import { createRequire } from 'node:module';
    import { readFileSync } from 'node:fs';

    const __dirname = path.dirname(fileURLToPath(import.meta.url));
    export const IOF_XSD_PATH = path.resolve(__dirname, './IOF.xsd');

    export interface XsdError { line?: number; column?: number; message: string; }

    let cachedSchema: unknown = null;
    function getSchema() {
      if (cachedSchema) return cachedSchema;
      const require = createRequire(import.meta.url);
      // Try libxmljs2-xsd first; fall back to xmllint-wasm.
      try {
        const xsd = require('libxmljs2-xsd');
        cachedSchema = xsd.parseFile(IOF_XSD_PATH);
        return cachedSchema;
      } catch (e) {
        // Fallback: xmllint-wasm pre-loaded with the IOF.xsd content.
        const { xmllint } = require('xmllint-wasm');
        const xsdContent = readFileSync(IOF_XSD_PATH, 'utf8');
        cachedSchema = { validate: (xml: string) => xmllint.validateXML({ xml: [{ fileName: 'in.xml', contents: xml }], schema: [xsdContent] }) };
        return cachedSchema;
      }
    }

    export function validateXml(xmlSource: string): { valid: true } | { valid: false; errors: XsdError[] } {
      const schema = getSchema() as any;
      const errors = schema.validate(xmlSource);
      if (!errors || (Array.isArray(errors) &amp;&amp; errors.length === 0) || errors.errors?.length === 0) return { valid: true };
      // Normalize the two libraries' error shapes:
      const list: XsdError[] = Array.isArray(errors) ? errors.map((e: any) => ({ line: e.line, column: e.column, message: e.message })) :
        (errors.errors ?? []).map((e: any) => ({ line: e.loc?.lineNumber, message: e.message ?? String(e) }));
      return { valid: false, errors: list };
    }
    ```

    apps/edge/src/ingest/courseImport.ts — pure ingester:
    ```
    import { sql } from 'drizzle-orm';
    import crypto from 'node:crypto';
    import type { DbHandle } from '../db/index.ts';
    import { classes, controls, courses, course_controls } from '../db/schema.ts';
    import type { ParsedCourseData } from '../xml/parse.ts';

    export interface CourseImportResult {
      classes_created: number;
      controls_created: number;
      courses_created: number;
    }

    export function ingestCourseData(handle: DbHandle, competitionId: string, data: ParsedCourseData): CourseImportResult {
      let cls = 0, ctl = 0, crs = 0;
      handle.sqlite.transaction(() => {
        // 1. Auto-create classes by name (D-15 wizard: auto-create if not present, REQ-EVT-CMP-002).
        const classIdByName = new Map<string, string>();
        for (const c of data.classes) {
          const existing = handle.db.select().from(classes).where(sql`competition_id = ${competitionId} AND name = ${c.name}`).get();
          if (existing) { classIdByName.set(c.name, existing.id); continue; }
          const id = crypto.randomUUID();
          handle.db.insert(classes).values({ id, competitionId, name: c.name, shortName: c.short_name }).run();
          classIdByName.set(c.name, id);
          cls++;
        }
        // 2. Upsert controls by code.
        const controlIdByCode = new Map<number, string>();
        for (const ct of data.controls) {
          const existing = handle.db.select().from(controls).where(sql`competition_id = ${competitionId} AND code = ${ct.code}`).get();
          if (existing) { controlIdByCode.set(ct.code, existing.id); continue; }
          const id = crypto.randomUUID();
          handle.db.insert(controls).values({ id, competitionId, code: ct.code }).run();
          controlIdByCode.set(ct.code, id);
          ctl++;
        }
        // 3. Insert courses + course_controls.
        for (const cr of data.courses) {
          const id = crypto.randomUUID();
          const classId = cr.class_id_ref ? classIdByName.get(cr.class_id_ref) ?? null : null;
          handle.db.insert(courses).values({ id, competitionId, name: cr.name, classId, lengthM: cr.length_m, climbM: cr.climb_m }).run();
          crs++;
          for (let i = 0; i < cr.control_codes.length; i++) {
            const code = cr.control_codes[i]!;
            const controlId = controlIdByCode.get(code);
            if (!controlId) throw new Error(`Course ${cr.name} references unknown control ${code}`);
            handle.db.insert(course_controls).values({ id: crypto.randomUUID(), courseId: id, controlId, orderIdx: i }).run();
          }
        }
      })();
      return { classes_created: cls, controls_created: ctl, courses_created: crs };
    }
    ```

    export function ingestEntryList(handle: DbHandle, competitionId: string, data: ParsedEntryList, consentAtMs: number): EntryImportResult {
      // Map class names to ids (must already exist; REQ-EVT-CMP-003 expects EntryList AFTER CourseData).
      // If a class is missing, skip those entries and return the list to surface "import partial" warning in UI.
      const classIdByName = new Map<string, string>();
      const existingClasses = handle.db.select().from(classes).where(sql`competition_id = ${competitionId}`).all();
      for (const cls of existingClasses) classIdByName.set(cls.name, cls.id);
      const missing = new Set<string>();
      let n = 0;
      handle.sqlite.transaction(() => {
        for (const e of data.competitors) {
          const classId = classIdByName.get(e.class_name);
          if (!classId) { missing.add(e.class_name); continue; }
          // Check partial unique (card_number, competition_id) collision; skip + record warning.
          if (e.card_number !== null) {
            const dup = handle.db.select().from(competitors).where(sql`competition_id = ${competitionId} AND card_number = ${e.card_number}`).get();
            if (dup) continue; // skip dup; do not block import
          }
          handle.db.insert(competitors).values({
            id: crypto.randomUUID(), competitionId, name: e.name, club: e.club, classId, cardNumber: e.card_number,
            consentAtMs, scrubbedAtMs: null,
          }).run();
          // Upsert club autocomplete.
          if (e.club) {
            handle.db.insert(clubs).values({ name: e.club, lastSeenAtMs: consentAtMs })
              .onConflictDoUpdate({ target: clubs.name, set: { lastSeenAtMs: consentAtMs } }).run();
          }
          n++;
        }
      })();
      return { competitors_created: n, classes_missing: [...missing] };
    }

    import { isAbsolute } from 'node:path';
    import { parseIofXml } from '../xml/parse.ts';
    import { validateXml } from '../xml/validate.ts';
    import { ingestCourseData } from '../ingest/courseImport.ts';
    import { ingestEntryList } from '../ingest/entryImport.ts';
    import multipart from '@fastify/multipart';

    export default async function importRoutes(app: FastifyInstance) {
      await app.register(multipart, { limits: { fileSize: 5 * 1024 * 1024, files: 1 } });

      app.post('/api/competitions/:id/import', async (req, reply) => {
        const competitionId = (req.params as { id: string }).id;
        const part = await req.file();
        if (!part) return reply.code(400).send({ error: 'no_file', message: 'Förväntar en fil.' });
        // Sanitize filename — T-PATH-TRAVERSAL: store only the bytes; reject filenames with '..' or absolute paths.
        if (part.filename.includes('..') || isAbsolute(part.filename)) return reply.code(400).send({ error: 'bad_filename' });
        const bytes = await part.toBuffer();
        const xmlSource = bytes.toString('utf8');

        // Pre-parse safety + dispatch.
        let parsed;
        try { parsed = parseIofXml(xmlSource); } catch (e) {
          return reply.code(400).send({ error: 'parse_failed', message: 'Filen kunde inte läsas — förväntar Purple Pen .xml eller IOF XML 3.0 CourseData.' , detail: (e as Error).message });
        }

        // XSD validation BEFORE any DB write (SC#6 binding contract).
        const validation = validateXml(xmlSource);
        if (!validation.valid) {
          return reply.code(400).send({ error: 'xsd_invalid', errors: validation.errors });
        }

        // Verify competition exists.
        const comp = app.fartolDb.db.select().from(competitions).where(eq(competitions.id, competitionId)).get();
        if (!comp) return reply.code(404).send({ error: 'competition_not_found' });

        if (parsed.kind === 'CourseData') {
          const result = ingestCourseData(app.fartolDb, competitionId, parsed.data);
          return reply.code(201).send({ kind: 'CourseData', ...result });
        } else {
          const result = ingestEntryList(app.fartolDb, competitionId, parsed.data, Date.now());
          return reply.code(201).send({ kind: 'EntryList', ...result });
        }
      });
    }
