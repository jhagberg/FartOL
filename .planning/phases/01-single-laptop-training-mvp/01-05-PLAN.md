**C-H3 atomic-wizard endpoint:**

- POST /api/competitions/from-wizard { name, date, xml_file: { name, content_base64 } } → 201 with { competition_id, kind, ...ingestResult } OR 400/422 with error.
- Body schema (Zod, lands in plan 04's competitions schema barrel or in this plan's route file):
  - `name`: string, min 1.
  - `date`: string matching /^\d{4}-\d{2}-\d{2}$/.
  - `xml_file.name`: string, sanitized (no '..', not absolute).
  - `xml_file.content_base64`: string, base64-decoded into bytes ≤ 5 MB.
- Handler flow (single sqlite.transaction):
  1. Decode base64 → bytes → UTF-8 string.
  2. Pre-parse: parseIofXml() (DOCTYPE/ENTITY pre-flight + root dispatch). On throw → 400 'parse_failed' with no DB writes.
  3. XSD: validateXml(xmlSource). On invalid → 400 'xsd_invalid' with errors[].
  4. Open `handle.sqlite.transaction(() => { ... })()`:
     - INSERT into competitions ({ id: uuidv4, name, date, receipt_template: 'classic', auto_print: 0, created_at_ms: Date.now() }). Capture the generated id.
     - Call ingestCourseData(handle, competitionId, parsed.data, { outerTransaction: true }) OR ingestEntryList(handle, competitionId, parsed.data, Date.now(), { outerTransaction: true }) depending on parsed.kind.
  5. If the transaction throws (ingest error, FK violation, etc.), better-sqlite3 rolls it back — no competition row persisted.
  6. Return 201 { competition_id, kind, ...ingestResult }.
- For EntryList-only imports via this endpoint: competition is created EMPTY of classes/courses; competitor inserts will fail their class FK lookup; the executor surfaces this as a 422 'entrylist_without_courses' (the wizard normally imports CourseData first, but the endpoint validates the precondition). Plan 12 wizard always uploads CourseData via this endpoint.

T-PATH-TRAVERSAL: the from-wizard endpoint accepts `xml_file.name` as metadata only — used only for the audit log / error messages. Bytes flow via `xml_file.content_base64`. The endpoint never writes the filename to disk. Still sanitize: reject if filename contains '..' OR is an absolute path.

    ```
    import { XMLParser } from 'fast-xml-parser';

    export type ParsedXml =
      | { kind: 'CourseData'; data: ParsedCourseData }
      | { kind: 'EntryList'; data: ParsedEntryList };

    const parser = new XMLParser({
      ignoreAttributes: false,
      attributeNamePrefix: '@',
      processEntities: false,         // T-FILE-IMPORT: disable entity expansion
      allowBooleanAttributes: true,
    });

    export function parseIofXml(xmlSource: string): ParsedXml {
      // Pre-flight: reject DOCTYPE and external entity declarations.
      if (/<!DOCTYPE/i.test(xmlSource)) throw new Error('DOCTYPE not allowed');
      if (/<!ENTITY/i.test(xmlSource)) throw new Error('ENTITY declarations not allowed');
      const parsed = parser.parse(xmlSource);
      const rootKey = Object.keys(parsed).find(k => k !== '?xml' &amp;&amp; k !== '@_xmlns');
      if (!rootKey) throw new Error('No XML root element');
      if (rootKey === 'CourseData') return { kind: 'CourseData', data: normalizeCourseData(parsed.CourseData) };
      if (rootKey === 'EntryList') return { kind: 'EntryList', data: normalizeEntryList(parsed.EntryList) };
      throw new Error(`Unsupported XML root element: ${rootKey}. Expected CourseData (Purple Pen, IOF 3.0) or EntryList (IOF 3.0).`);
    }
    ```

    ```
    import { classes, controls, courses, course_controls } from '../db/schema.ts';

    export interface CourseImportResult { classes_created: number; controls_created: number; courses_created: number; }
    export interface CourseImportOpts { outerTransaction?: boolean; }

    function doIngest(handle: DbHandle, competitionId: string, data: ParsedCourseData): CourseImportResult {
      let cls = 0, ctl = 0, crs = 0;
      const classIdByName = new Map<string, string>();
      for (const c of data.classes) {
        const existing = handle.db.select().from(classes).where(sql`competition_id = ${competitionId} AND name = ${c.name}`).get();
        if (existing) { classIdByName.set(c.name, existing.id); continue; }
        const id = crypto.randomUUID();

        if (existing) { controlIdByCode.set(ct.code, existing.id); continue; }
        const id = crypto.randomUUID();
        handle.db.insert(controls).values({ id, competitionId, code: ct.code }).run();
        controlIdByCode.set(ct.code, id); ctl++;
      }

        const id = crypto.randomUUID();
        const classId = cr.class_id_ref ? classIdByName.get(cr.class_id_ref) ?? null : null;
        handle.db.insert(courses).values({ id, competitionId, name: cr.name, classId, lengthM: cr.length_m, climbM: cr.climb_m }).run();
        crs++;
        for (let i = 0; i < cr.control_codes.length; i++) {
          const code = cr.control_codes[i]!;
          const controlId = controlIdByCode.get(code);
          if (!controlId) throw new Error(`Course ${cr.name} references unknown control ${code}`);
          handle.db.insert(course_controls).values({ id: crypto.randomUUID(), courseId: id, controlId, orderIdx: i }).run();
        }
      }
      return { classes_created: cls, controls_created: ctl, courses_created: crs };

      if (opts.outerTransaction) return doIngest(handle, competitionId, data);
      let result: CourseImportResult = { classes_created: 0, controls_created: 0, courses_created: 0 };
      handle.sqlite.transaction(() => { result = doIngest(handle, competitionId, data); })();
      return result;
    }
    ```

    ```
    // Authored for fartola. Not ported from upstream.
    // POST /api/competitions/from-wizard — atomic competition create + XML import.
    // Per codex review C-H3: HTTP requests cannot share a SQL transaction. The
    // wizard's "no orphan competition rows" contract requires a single endpoint
    // that wraps both INSERT competition + ingest in ONE sqlite.transaction so
    // an XML-validation or ingest failure rolls back the competition INSERT.
    import { z } from 'zod';
    import crypto from 'node:crypto';
    import { isAbsolute } from 'node:path';
    import type { FastifyInstance } from 'fastify';
    import { competitions } from '../db/schema.ts';
    import { parseIofXml } from '../xml/parse.ts';
    import { validateXml } from '../xml/validate.ts';
    import { ingestCourseData } from '../ingest/courseImport.ts';
    import { ingestEntryList } from '../ingest/entryImport.ts';

    const Schema = z.object({
      name: z.string().min(1).max(200),
      date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
      xml_file: z.object({
        name: z.string().min(1).max(255),
        content_base64: z.string().min(1),
      }),
    });

    export default async function competitionsFromWizardRoutes(app: FastifyInstance) {
      app.post('/api/competitions/from-wizard', async (req, reply) => {
        const parsed = Schema.safeParse(req.body);
        if (!parsed.success) return reply.code(400).send({ error: 'bad_request', issues: parsed.error.issues });
        const { name, date, xml_file } = parsed.data;
        if (xml_file.name.includes('..') || isAbsolute(xml_file.name)) return reply.code(400).send({ error: 'bad_filename' });
        let xmlSource: string;
        try {
          const bytes = Buffer.from(xml_file.content_base64, 'base64');
          if (bytes.byteLength > 5 * 1024 * 1024) return reply.code(413).send({ error: 'file_too_large' });
          xmlSource = bytes.toString('utf8');
        } catch {
          return reply.code(400).send({ error: 'bad_base64' });
        }

        let parsedXml;
        try { parsedXml = parseIofXml(xmlSource); } catch (e) {
          return reply.code(400).send({ error: 'parse_failed', detail: (e as Error).message });
        }
        const validation = validateXml(xmlSource);
        if (!validation.valid) return reply.code(400).send({ error: 'xsd_invalid', errors: validation.errors });

        // C-H3 LOCKED: single sqlite.transaction wraps INSERT competition + ingest.
        // Any throw inside the transaction rolls back the entire unit — no orphan rows.
        const competitionId = crypto.randomUUID();
        try {
          let ingestResult: unknown;
          let kind: 'CourseData' | 'EntryList';
          app.fartolaDb.sqlite.transaction(() => {
            app.fartolaDb.db.insert(competitions).values({
              id: competitionId, name, date,
              receiptTemplate: 'classic', autoPrint: false, createdAtMs: Date.now(),
            }).run();
            if (parsedXml.kind === 'CourseData') {
              kind = 'CourseData';
              ingestResult = ingestCourseData(app.fartolaDb, competitionId, parsedXml.data, { outerTransaction: true });
            } else {
              kind = 'EntryList';
              ingestResult = ingestEntryList(app.fartolaDb, competitionId, parsedXml.data, Date.now(), { outerTransaction: true });
            }
          })();
          return reply.code(201).send({ competition_id: competitionId, kind: kind!, ...(ingestResult as object) });
        } catch (e) {
          app.log.warn({ err: (e as Error).message }, 'from-wizard transaction rolled back');
          return reply.code(422).send({ error: 'ingest_failed', detail: (e as Error).message });
        }
      });
    }
    ```
