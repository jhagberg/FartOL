WS envelope shapes (LOCKED here for results channel):
- `{ type: 'results_update'; channel: 'results:<id>'; payload: { class_id: string; class_name: string; rows: ResultView[] }; seq: number }` — sent per-class when that class's rows changed.
- `{ type: 'results_full'; channel: 'results:<id>'; payload: { classes: Array<{ class_id: string; class_name: string; rows: ResultView[] }>; pending_unknown_cards: number[] }; seq: number }` — sent ONCE on hello as the full state replay (cheaper than re-walking events).

Debounce rationale: a real Tuesday training never produces > 1 card_read/sec, but the reducer is fast (< 200ms for 1000 events per plan 07). Debounce 50ms protects against bursts (operator clicks a button that triggers 10 simulate-reads in rapid succession during dev).

    export function loadCompetitionInputs(handle: DbHandle, competitionId: string): ReduceInput | null {
      const competition = handle.db.select().from(competitions).where(eq(competitions.id, competitionId)).get();
      if (!competition) return null;
      const competitorsRows = handle.db.select().from(competitors).where(eq(competitors.competitionId, competitionId)).all();
      const classesRows = handle.db.select().from(classes).where(eq(classes.competitionId, competitionId)).all();
      // Eager-load courses with control_codes ordered by order_idx.
      const coursesRows = handle.db.select().from(courses).where(eq(courses.competitionId, competitionId)).all();
      const coursesWithCodes = coursesRows.map(c => {
        const codes = handle.db
          .select({ code: controls.code, orderIdx: course_controls.orderIdx })
          .from(course_controls).innerJoin(controls, eq(course_controls.controlId, controls.id))
          .where(eq(course_controls.courseId, c.id))
          .orderBy(asc(course_controls.orderIdx))
          .all();
        return { ...c, control_codes: codes.map(r => r.code) };
      });
      // Events: pull all for the competition. For 20-40 starters x ~30 events ea = ~1200 rows max; not paginated.
      const eventsRows = handle.db.select().from(events).where(eq(events.competitionId, competitionId)).orderBy(asc(events.eventTimeMs), asc(events.localSeq)).all();
      return { competition_id: competitionId, events: eventsRows, competitors: competitorsRows, classes: classesRows, courses: coursesWithCodes };
    }

    apps/edge/src/projection/store.ts:
    ```
    import { reduce } from './reduce.ts';
    import { loadCompetitionInputs } from './loader.ts';
    import type { DbHandle } from '../db/index.ts';
    import type { CompetitionState } from './types.ts';
    import { resultsChannel, type ChannelName } from '@fartol/shared-types';

    export interface ProjectionStore {
      get(competitionId: string): CompetitionState | null;
      markDirty(competitionId: string): void;
      recomputeNow(competitionId: string): CompetitionState | null;
      dispose(): void;
    }

    export function createProjectionStore(opts: {
      handle: DbHandle;
      broadcast: (channel: ChannelName, envelope: { type: string; payload: unknown; seq: number }) => void;
      debounceMs?: number;
    }): ProjectionStore {
      const debounceMs = opts.debounceMs ?? 50;
      const cache = new Map<string, CompetitionState>();
      const pending = new Map<string, NodeJS.Timeout>();
      let disposed = false;

      function recomputeNow(competitionId: string): CompetitionState | null {
        const input = loadCompetitionInputs(opts.handle, competitionId);
        if (!input) return null;
        const previous = cache.get(competitionId);
        const next = reduce(input);
        cache.set(competitionId, next);
        // Broadcast per-class results_update for classes whose rows changed. For Phase 1 simplicity, broadcast all classes (cheap for 20-40 starters).
        for (const [classId, rows] of next.results_by_class) {
          const cls = input.classes.find(c => c.id === classId);
          opts.broadcast(resultsChannel(competitionId), { type: 'results_update', payload: { class_id: classId, class_name: cls?.name ?? '', rows }, seq: next.last_event_seq });
        }
        return next;
      }

      return {
        get(id) { return cache.get(id) ?? null; },
        markDirty(id) {
          if (disposed) return;
          if (pending.has(id)) return;
          const handle = setTimeout(() => { pending.delete(id); recomputeNow(id); }, debounceMs);
          pending.set(id, handle);
        },
        recomputeNow,
        dispose() { disposed = true; for (const t of pending.values()) clearTimeout(t); pending.clear(); },
      };
    }
    ```

    export default async function resultsRoutes(app: FastifyInstance) {
      app.get('/api/competitions/:id/results', async (req, reply) => {
        const competitionId = (req.params as { id: string }).id;
        let state = app.projectionStore.get(competitionId);
        if (!state) state = app.projectionStore.recomputeNow(competitionId);
        if (!state) return reply.code(404).send({ error: 'competition_not_found' });
        // Pull class metadata for names.
        const classesList = app.fartolDb.db.select().from(classes).where(eq(classes.competitionId, competitionId)).all();
        const byClass = classesList.map(c => ({ class_id: c.id, class_name: c.name, rows: state!.results_by_class.get(c.id) ?? [] }));
        return reply.send({ classes: byClass, pending_unknown_cards: state.pending_unknown_cards, last_event_seq: state.last_event_seq });
      });
    }
