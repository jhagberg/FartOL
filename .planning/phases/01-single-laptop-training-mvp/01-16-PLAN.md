XML shape (RESEARCH §Pattern 7 + Pitfall 5 conservative subset). Top-level:
```
<?xml version="1.0" encoding="UTF-8"?>
<ResultList xmlns="http://www.orienteering.org/datastandard/3.0" iofVersion="3.0" createTime="<ISO>" creator="FartOL v0.1">
  <Event>
    <Name>{competition.name}</Name>
    <StartTime><Date>{competition.date}</Date></StartTime>
  </Event>
  <!-- Per class -->
  <ClassResult>
    <Class><Name>{class.name}</Name></Class>
    <!-- Per competitor -->
    <PersonResult>
      <Person>
        <Name><Family>{family}</Family><Given>{given}</Given></Name>
      </Person>
      <Organisation type="Club"><Name>{club}</Name></Organisation>   <!-- only if club non-null -->
      <Result>
        <StartTime>{ISO start}</StartTime>                            <!-- only if known -->
        <FinishTime>{ISO finish}</FinishTime>                          <!-- only if known -->
        <Time>{elapsed seconds}</Time>                                 <!-- integer seconds -->
        <Position>{place}</Position>                                   <!-- only if status=OK and place set -->
        <Status>{OK|MissingPunch|DidNotFinish}</Status>
      </Result>
    </PersonResult>
  </ClassResult>
  <!-- Provisional status only when input.status === 'Provisional' -->
</ResultList>
```

Status map per IOF spec:
- 'OK' → 'OK'
- 'MP' → 'MissingPunch'
- 'DNF' → 'DidNotFinish'
- 'PEND' → omitted (don't emit PersonResult for pending starters? — Phase 1 LOCKED: emit them with <Status>NotStarted</Status> per IOF.xsd enum; document the choice).

DECISION (LOCKED here): PEND competitors are EXCLUDED from the export entirely. ResultList semantics target finished events; pending starters get re-exported when they finish. Phase 2 operator-controlled "include not-started" flag can land later. The provisional toggle does NOT change this; provisional means "results are final-for-now, may be amended."

Name splitting: competitor.name is one string "Given Family"; split on last space (Swedish naming convention: last token is family). If only one token, Family=token, Given=''. The XSD requires both `<Family>` AND `<Given>` per the v3.0 schema; emit empty `<Given/>` if needed.

Time format: IOF XML 3.0 ResultList `<Time>` is decimal seconds (per the XSD's xsd:decimal). E.g., 12:30 elapsed → `<Time>750</Time>`. `<StartTime>` and `<FinishTime>` are xsd:dateTime ISO 8601 (combine competition.date with the wall-clock derived from competitor's start_time_ms relative to local timezone — for Phase 1, derive from event_time_ms as a Date.toISOString()).

Export endpoint shapes:
- GET /api/competitions/:id/export/preview?status=Final|Provisional → 200 { valid: true, summary: { class_count, person_result_count, status } } OR { valid: false, errors: XsdError[] }.
- GET /api/competitions/:id/export?format=iof30&amp;status=Final|Provisional → 200 with Content-Type: application/xml + Content-Disposition: attachment; filename="<slug>-resultlist.xml" — the XML body. Validation runs first; on fail, return 400 with errors[] (NOT a partial file).

    ```
    import { XMLBuilder } from 'fast-xml-parser';
    import { validateXml } from './validate.ts';
    import type { CompetitionState, CompetitorView, ResultView } from '../projection/types.ts';

    export interface ExportInput { competition: CompetitionDTO; classes: ClassDTO[]; courses: CourseDTO[]; state: CompetitionState; status?: 'Final' | 'Provisional'; creator?: string; }
    export interface ExportSummary { class_count: number; person_result_count: number; status: 'Final' | 'Provisional'; }
    export interface BuildResult { xml: string; summary: ExportSummary; }
    export interface ValidatedBuildResult { valid: boolean; errors?: import('./validate.ts').XsdError[]; build?: BuildResult; }

    function splitName(full: string): { family: string; given: string } { const trimmed = full.trim(); const idx = trimmed.lastIndexOf(' '); if (idx < 0) return { family: trimmed, given: '' }; return { family: trimmed.slice(idx + 1), given: trimmed.slice(0, idx) }; }
    function statusForXml(s: CompetitorView['status']): string | null { return ({ OK: 'OK', MP: 'MissingPunch', DNF: 'DidNotFinish', PEND: null } as const)[s]; }
    function buildPersonResult(v: CompetitorView): unknown | null {
      const xmlStatus = statusForXml(v.status); if (!xmlStatus) return null;  // PEND skipped
      const { family, given } = splitName(v.name);
      const pr: any = { Person: { Name: { Family: family, Given: given || { '#text': '' } } } };
      if (v.club) pr.Organisation = { '@_type': 'Club', Name: v.club };
      const res: any = { Status: xmlStatus };
      // Inject start/finish from latest read history (plan 07 stores card_read_history with event_time_ms + punches; for export we use the StatusResult-derived fields stored on the view — plan 07 carries start_time_ms / finish_time_ms via the projection if we add them; if not, recompute from latest_punches here using the dnfMp module).
      // SIMPLIFICATION: trust v.elapsed_time_ms; emit Time = seconds. Skip StartTime/FinishTime in P1 (operator-acceptable per RESEARCH Pitfall 5 conservative subset).
      if (v.elapsed_time_ms !== null) res.Time = Math.floor(v.elapsed_time_ms / 1000);
      if (v.status === 'OK') {
        // Place comes from the per-class ResultView, NOT from CompetitorView directly. Pull from input.state.results_by_class.
        // The caller injects place: handled in buildResultListXml.
      }
      pr.Result = res;
      return pr;
    }

    export function buildResultListXml(input: ExportInput): BuildResult {
      const status = input.status ?? 'Final';
      const creator = input.creator ?? 'FartOL v0.1';
      const classResults: unknown[] = [];
      let personResultCount = 0;
      for (const cls of input.classes) {
        const rows = input.state.results_by_class.get(cls.id) ?? [];
        const personResults: any[] = [];
        for (const row of rows) {
          const view = input.state.competitors.get(row.competitor_id);
          if (!view) continue;
          const pr = buildPersonResult(view) as any;
          if (!pr) continue;
          if (view.status === 'OK' &amp;&amp; row.place !== null) pr.Result.Position = row.place;
          personResults.push(pr);
          personResultCount++;
        }
        if (personResults.length === 0) continue;
        classResults.push({ Class: { Name: cls.name }, PersonResult: personResults });
      }
      const tree = {
        '?xml': { '@_version': '1.0', '@_encoding': 'UTF-8' },
        ResultList: {
          '@_xmlns': 'http://www.orienteering.org/datastandard/3.0',
          '@_iofVersion': '3.0',
          '@_createTime': new Date().toISOString(),
          '@_creator': creator,
          ...(status === 'Provisional' ? { '@_status': 'Snapshot' } : {}),  // IOF 3.0 ResultList allows @status="Complete"|"Delta"|"Snapshot"|"Final"
          Event: { Name: input.competition.name, StartTime: { Date: input.competition.date } },
          ClassResult: classResults,
        },
      };
      const builder = new XMLBuilder({ ignoreAttributes: false, attributeNamePrefix: '@_', format: true, indentBy: '  ', suppressEmptyNode: false });
      const xml = builder.build(tree);
      return { xml, summary: { class_count: classResults.length, person_result_count: personResultCount, status } };
    }

    export function validateAndBuild(input: ExportInput): ValidatedBuildResult {
      const built = buildResultListXml(input);
      const v = validateXml(built.xml);
      if (!v.valid) return { valid: false, errors: v.errors };
      return { valid: true, build: built };
    }
    ```

    ```
    import { validateAndBuild } from '../xml/iofExport.ts';
    import { eq } from 'drizzle-orm';
    import { competitions, classes as classesTable } from '../db/schema.ts';
    import { loadCompetitionInputs } from '../projection/loader.ts';

    export default async function exportRoutes(app: FastifyInstance) {
      app.get('/api/competitions/:id/export/preview', async (req, reply) => {
        const { id } = req.params as { id: string };
        const status = (req.query as any).status === 'Provisional' ? 'Provisional' : 'Final';
        const input = loadCompetitionInputs(app.fartolDb, id);
        if (!input) return reply.code(404).send({ error: 'competition_not_found' });
        const comp = app.fartolDb.db.select().from(competitions).where(eq(competitions.id, id)).get()!;
        const classes = app.fartolDb.db.select().from(classesTable).where(eq(classesTable.competitionId, id)).all();
        const state = app.projectionStore.recomputeNow(id)!;
        const result = validateAndBuild({ competition: { ...comp, auto_print: !!comp.autoPrint, receipt_template: comp.receiptTemplate, created_at_ms: comp.createdAtMs } as any, classes: classes.map(c => ({ ...c, competition_id: c.competitionId, short_name: c.shortName })) as any, courses: input.courses as any, state, status });
        if (!result.valid) return reply.code(200).send({ valid: false, errors: result.errors });
        return reply.send({ valid: true, summary: result.build!.summary });
      });

      app.get('/api/competitions/:id/export', async (req, reply) => {
        const { id } = req.params as { id: string };
        const format = (req.query as any).format;
        const status = (req.query as any).status === 'Provisional' ? 'Provisional' : 'Final';
        if (format !== 'iof30') return reply.code(400).send({ error: 'unsupported_format' });
        const input = loadCompetitionInputs(app.fartolDb, id);
        if (!input) return reply.code(404).send({ error: 'competition_not_found' });
        const comp = app.fartolDb.db.select().from(competitions).where(eq(competitions.id, id)).get()!;
        const classes = app.fartolDb.db.select().from(classesTable).where(eq(classesTable.competitionId, id)).all();
        const state = app.projectionStore.recomputeNow(id)!;
        const result = validateAndBuild({ competition: { ...comp, auto_print: !!comp.autoPrint, receipt_template: comp.receiptTemplate, created_at_ms: comp.createdAtMs } as any, classes: classes.map(c => ({ ...c, competition_id: c.competitionId, short_name: c.shortName })) as any, courses: input.courses as any, state, status });
        if (!result.valid) return reply.code(400).send({ error: 'xsd_invalid', errors: result.errors });
        const slug = comp.name.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '') || 'competition';
        reply.header('Content-Type', 'application/xml; charset=utf-8');
        reply.header('Content-Disposition', `attachment; filename="${slug}-resultlist.xml"`);
        return reply.send(result.build!.xml);
      });
    }
    ```
