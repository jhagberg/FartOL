    ```
    // Authored for fartol. Not ported from upstream.
    // Opens a better-sqlite3 connection, sets WAL pragmas, runs the embedded
    // migrator, returns the drizzle handle + the raw sqlite + a close() helper.
    // Per .planning/phases/01-single-laptop-training-mvp/01-RESEARCH.md §Pattern 2.
    import Database from 'better-sqlite3';
    import { drizzle } from 'drizzle-orm/better-sqlite3';
    import * as schema from './schema.ts';
    import { runMigrations } from './migrate.ts';

    export interface DbHandle {
      db: ReturnType<typeof drizzle<typeof schema>>;
      sqlite: Database.Database;
      close: () => void;
    }
    export function openDatabase(dbPath: string): DbHandle {
      const sqlite = new Database(dbPath);
      sqlite.pragma('journal_mode = WAL');
      sqlite.pragma('synchronous = NORMAL');
      sqlite.pragma('foreign_keys = ON');
      sqlite.pragma('cache_size = -32000');
      sqlite.pragma('temp_store = MEMORY');
      runMigrations(sqlite);
      const db = drizzle(sqlite, { schema });
      return { db, sqlite, close: () => sqlite.close() };
    }
    ```

    ```
    // Embedded Drizzle migrator. Runs at every cold start; idempotent.
    // migrationsFolder resolves relative to this file so the published tarball
    // ships drizzle/ alongside dist/db/migrate.cjs and the migrator finds it.

    import { fileURLToPath } from 'node:url';
    import type Database from 'better-sqlite3';
    import { drizzle } from 'drizzle-orm/better-sqlite3';
    ```

    apps/edge/src/db/node-id.ts — UUID persistence:
    ```
    import { config } from './schema.ts';
    import { eq } from 'drizzle-orm';
    import crypto from 'node:crypto';
    import type { DbHandle } from './index.ts';

    export function ensureNodeId(handle: DbHandle): string {
      const existing = handle.db.select().from(config).where(eq(config.key, 'node_id')).get();
      if (existing) return existing.value;
      const nodeId = crypto.randomUUID();
      handle.db.insert(config).values({ key: 'node_id', value: nodeId }).run();
      return nodeId;
    }
    ```

    apps/edge/src/db/seq.ts — local_seq next-value helper:
    ```
    import { events } from './schema.ts';
    import { sql, eq, max } from 'drizzle-orm';
    import type { DbHandle } from './index.ts';

    export function nextLocalSeq(handle: DbHandle, nodeId: string): number {
      const row = handle.db
        .select({ max: sql<number>`coalesce(max(${events.localSeq}), 0)` })
        .from(events)
        .where(eq(events.nodeId, nodeId))
        .get();
      return (row?.max ?? 0) + 1;
    }
    ```
