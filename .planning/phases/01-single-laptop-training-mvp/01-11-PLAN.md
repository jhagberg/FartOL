Svelte 5 runes store pattern (CONTEXT D-13 + UI-SPEC notes — operator preferences):
```
// apps/web/src/lib/stores/tweaks.svelte.ts
import { browser } from '$app/environment';
const STORAGE_KEY = 'fartol.tweaks.v1';
export interface TweaksState {
  locale: 'sv' | 'en';
  density: 'low' | 'med' | 'high';
  accent: 'forest' | 'blue' | 'magenta' | 'charcoal';
  contrast_high: boolean;
  font_pair: 'plex' | 'geist' | 'source' | 'atkinson';
}
function defaultState(): TweaksState { return { locale: 'sv', density: 'med', accent: 'forest', contrast_high: false, font_pair: 'plex' }; }
function load(): TweaksState { if (!browser) return defaultState(); try { return { ...defaultState(), ...JSON.parse(localStorage.getItem(STORAGE_KEY) ?? '{}') }; } catch { return defaultState(); } }
export const tweaks = $state<TweaksState>(load());
export function persistTweaks() { if (browser) localStorage.setItem(STORAGE_KEY, JSON.stringify(tweaks)); }
```
Note: Svelte 5 runes (`$state`) outside a component require the `.svelte.ts` extension (the SvelteKit + vite plugin compiles them).

// apps/web/src/lib/i18n/index.ts
import i18next from 'i18next';
import sv from './sv.json';
import en from './en.json';
import { tweaks } from '../stores/tweaks.svelte.ts';
i18next.init({ lng: tweaks.locale, fallbackLng: 'en', interpolation: { escapeValue: false }, resources: { sv: { translation: sv }, en: { translation: en } } });
export const t = (key: string, vars?: Record<string, unknown>) => i18next.t(key, vars);
export function setLocale(locale: 'sv' | 'en') { i18next.changeLanguage(locale); tweaks.locale = locale; persistTweaks(); }
