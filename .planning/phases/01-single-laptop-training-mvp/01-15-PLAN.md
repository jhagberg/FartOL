    import { existsSync } from 'node:fs';
    import { createRequire } from 'node:module';
    import type { PrinterSink, PrintEnvelope } from './sink.ts';
    import { renderTemplate } from './templates.ts';

    export interface DriverOpts {
      printerType?: 'star' | 'epson' | 'brother';        // default 'star'
      devicePath?: string;                               // default: probe /dev/usb/lp0..3
      characterSet?: string;                             // default 'PC852_LATIN2' (covers sv å/ä/ö)
      queueCap?: number;                                 // default 50
      // Test injection (PATTERNS S-2):
      printerFactory?: (opts: { type: string; interface: string; characterSet: string }) => InternalThermalPrinter;
    }
    interface InternalThermalPrinter {
      isPrinterConnected(): Promise<boolean>;
      println(text: string): void;
      newLine(): void;
      bold(on: boolean): void;
      alignLeft(): void; alignCenter(): void; alignRight(): void;
      drawLine(): void;
      cut(): void;
      printImage(pathOrBuffer: string | Buffer): Promise<void>;
      execute(): Promise<void>;
      clear(): void;
    }

    export function createNodeThermalPrinterSink(opts: DriverOpts = {}): PrinterSink &amp; { dispose: () => void } {
      const queue: Array<() => Promise<void>> = [];
      let processing = false;
      let disposed = false;

      function probePath(): string | null {
        if (opts.devicePath) return opts.devicePath;
        for (let i = 0; i < 4; i++) { const p = `/dev/usb/lp${i}`; if (existsSync(p)) return p; }
        return null;
      }

      function buildPrinter(): InternalThermalPrinter | null {
        const path = probePath();
        if (!path) return null;
        if (opts.printerFactory) return opts.printerFactory({ type: opts.printerType ?? 'star', interface: `printer:${path}`, characterSet: opts.characterSet ?? 'PC852_LATIN2' });
        const require = createRequire(import.meta.url);
        const { printer: ThermalPrinter, types: Types } = require('node-thermal-printer');
        const typeMap = { star: Types.STAR, epson: Types.EPSON, brother: Types.BROTHER };
        return new ThermalPrinter({ type: typeMap[opts.printerType ?? 'star'], interface: `printer:${path}`, characterSet: opts.characterSet ?? 'PC852_LATIN2', removeSpecialCharacters: false, options: { timeout: 5000 } });
      }

      async function pump() {
        if (processing || disposed || queue.length === 0) return;
        processing = true;
        const job = queue.shift()!;
        try { await job(); } catch (e) { /* swallow — already surfaced via reject */ }
        processing = false;
        if (!disposed) pump();
      }

      return {
        async isPrinterConnected() {
          const printer = buildPrinter();
          if (!printer) return false;
          return printer.isPrinterConnected();
        },
        print(envelope: PrintEnvelope): Promise<void> {
          if (disposed) return Promise.reject(new Error('disposed'));
          if (queue.length >= (opts.queueCap ?? 50)) return Promise.reject(new Error('queue_full'));
          return new Promise((resolve, reject) => {
            queue.push(async () => {
              const printer = buildPrinter();
              if (!printer) { reject(new Error('printer_offline')); return; }
              try {
                const connected = await printer.isPrinterConnected();
                if (!connected) { reject(new Error('printer_offline')); return; }
                printer.clear();
                await renderTemplate(printer, envelope.template, envelope.data);
                printer.cut();
                await printer.execute();
                resolve();
              } catch (e) { reject(e); }
            });
            pump();
          });
        },
        dispose() { disposed = true; queue.length = 0; },
      };
    }

    import { generateSkogis, type SkogisDescriptor } from '@fartol/shared-types';
    import { createRequire } from 'node:module';

    export interface KidsInput { card_number: number; name: string; club: string | null; class_id: string; }
    export interface BitmapOutput { png: Buffer; width: number; height: number; }

    export function descriptorToSvgString(d: SkogisDescriptor): string {
      const [x, y, w, h] = d.viewBox;
      const elements = d.elements.map(el => {
        const attrs = Object.entries(el.attrs).map(([k, v]) => `${k}="${v}"`).join(' ');
        return `<${el.type} ${attrs}/>`;
      }).join('');
      // Mono-printable: force black strokes, paper fills.
      return `<svg xmlns="http://www.w3.org/2000/svg" viewBox="${x} ${y} ${w} ${h}" stroke="#000" stroke-width="1.5" fill="#fff">${elements}</svg>`;
    }

    export async function generateKidsBitmap(input: KidsInput): Promise<BitmapOutput> {
      const descriptor = generateSkogis(input);
      const svg = descriptorToSvgString(descriptor);
      const require = createRequire(import.meta.url);
      const sharp = require('sharp');
      const png = await sharp(Buffer.from(svg)).resize({ width: 384, fit: 'inside' }).png({ palette: true }).toBuffer();
      // 384 px = standard 80mm thermal column at 8 dots/mm.
      return { png, width: 384, height: 384 };
    }
